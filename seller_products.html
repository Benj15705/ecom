function openTab(evt, tabName) { const tabContents = document.getElementsByClassName('tab-content'); for (let i = 0; i < tabContents.length; i++) { tabContents[i].style.display = 'none'; } const tabButtons = document.getElementsByClassName('tab-button'); for (let i = 0; i < tabButtons.length; i++) { tabButtons[i].className = tabButtons[i].className.replace(' active', ''); } document.getElementById(tabName).style.display = 'block'; evt.currentTarget.className += ' active'; } // Set default tab document.addEventListener('DOMContentLoaded', () => { document.querySelector('.tab-button').click(); });
